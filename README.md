# Usage
### install
```npm i react-prompt-viewer```

### include
```javascript
import React from 'react'
import { prompt } from 'react-prompt-viewer'

class Component extends React.Component {
    
  render() {
    return <Prompt content="hello world"/>
  }
}
```
